### 2.2 异步爬虫实现

```python
# crawler/async_crawler.py
import aiohttp
import asyncio
from typing import List, Dict
from bs4 import BeautifulSoup
import json

class CommentCrawler:
    def __init__(self, source: str, config: Dict):
        self.source = source
        self.config = config
        self.session: aiohttp.ClientSession = None
        
    async def __aenter__(self):
        self.session = aiohttp.ClientSession(
            headers=self.config.get("headers", {}),
            timeout=aiohttp.ClientTimeout(total=30)
        )
        return self
        
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        if self.session:
            await self.session.close()
    
    async def fetch_nga_posts(self, forum_id: str, page: int = 1) -> List[Dict]:
        """获取NGA帖子列表"""
        url = f"{self.config['base_url']}/thread.php?fid={forum_id}&page={page}"
        
        async with self.session.get(url) as response:
            html = await response.text()
            soup = BeautifulSoup(html, 'html.parser')
            
            posts = []
            for post in soup.select('.topic')[:50]:  # 每页取前50条
                posts.append({
                    "id": post.get('data-tid'),
                    "title": post.select_one('.topic-title').text.strip(),
                    "author": post.select_one('.author').text.strip(),
                    "content": "",  # 需要单独请求详情页
                    "reply_count": int(post.select_one('.reply-count').text or 0),
                    "timestamp": post.select_one('.post-time').text.strip(),
                    "source": "nga",
                    "url": f"{self.config['base_url']}/read.php?tid={post.get('data-tid')}"
                })
            
            await asyncio.sleep(self.config['rate_limit'])
            return posts
    
    async def fetch_bilibili_comments(self, oid: str, next_page: int = 0) -> List[Dict]:
        """获取B站视频评论"""
        url = f"{self.config['base_url']}/x/v2/reply"
        params = {
            "oid": oid,  # 视频ID
            "type": 1,   # 视频类型
            "next": next_page,
            "ps": 20     # 每页数量
        }
        
        async with self.session.get(url, params=params) as response:
            data = await response.json()
            
            comments = []
            if data.get("data", {}).get("replies"):
                for reply in data["data"]["replies"]:
                    comments.append({
                        "id": reply["rpid"],
                        "content": reply["content"]["message"],
                        "author": reply["member"]["uname"],
                        "like_count": reply["like"],
                        "reply_count": reply["rcount"],
                        "timestamp": reply["ctime"],
                        "source": "bilibili",
                        "parent_id": reply.get("parent", 0)
                    })
            
            await asyncio.sleep(self.config['rate_limit'])
            return comments
    
    async def batch_fetch(self, tasks: List[Dict]) -> List[Dict]:
        """批量获取，控制并发"""
        semaphore = asyncio.Semaphore(5)  # 最大并发5
        
        async def fetch_with_limit(task):
            async with semaphore:
                if task["type"] == "nga":
                    return await self.fetch_nga_posts(task["forum_id"], task["page"])
                elif task["type"] == "bilibili":
                    return await self.fetch_bilibili_comments(task["oid"], task["page"])
                return []
        
        results = await asyncio.gather(*[fetch_with_limit(t) for t in tasks])
        return [item for sublist in results for item in sublist]


# 使用示例

        posts = await crawler.fetch_nga_posts("-7955747", page=1)

    def _compile_kaomoji_patterns(self) -> Dict[str, re.Pattern]:
        """编译颜文字正则模式"""
        return {
            "positive": re.compile(r'[（\(][\w\s]*[▽∀Д⊙ω｀´艹♡♥✧][\w\s]*[）\)]'),
            "negative": re.compile(r'[（\(][\w\s]*[‵□艹╯┻━┻][\w\s]*[）\)]'),
        }
    
    def preprocess(self, text: str) -> Tuple[str, Dict]:
        """
        主预处理函数
        
        Returns:
            processed_text: 处理后的文本
            metadata: 预处理元数据（用于后续分析）
        """
        metadata = {
            "original_length": len(text),
            "has_emoji": False,
            "has_kaomoji": False,
            "slang_replaced": [],
        }
        
        # 1. Emoji转语义标签
        text, emoji_info = self._convert_emoji(text)
        metadata["has_emoji"] = emoji_info["count"] > 0
        metadata["emoji_tags"] = emoji_info["tags"]
        
        # 2. 颜文字识别与分类
        text, kaomoji_info = self._convert_kaomoji(text)
        metadata["has_kaomoji"] = kaomoji_info["count"] > 0
        metadata["kaomoji_sentiment"] = kaomoji_info["sentiment"]
        
        # 3. 游戏黑话标准化
        text, slang_info = self._normalize_slang(text)
        metadata["slang_replaced"] = slang_info
        
        # 4. 重复标点压缩
        text = self._compress_punctuation(text)
        
        # 5. 特殊字符清理
        text = self._clean_special_chars(text)
        
        metadata["processed_length"] = len(text)
        
        return text, metadata

    def _convert_kaomoji(self, text: str) -> Tuple[str, Dict]:
        """颜文字识别与情感分类"""
        kaomoji_sentiment = {"positive": 0, "negative": 0}
        
        # 正面颜文字
        positive_kaomojis = [
            "(๑•̀ㅂ•́)و✧", "(｡♥‿♥｡)", "(◕‿◕✿)", "(ﾉ◕ヮ◕)ﾉ*:･ﾟ✧",
            "(✿ ♥‿♥)", "(◍•ᴗ•◍)❤", "(｡♥‿♥｡)", "(ﾉ◕ヮ◕)ﾉ*:･ﾟ✧",
            "(✧ω✧)", "(◠‿◠✿)", "(◕ᴗ◕✿)", "(｡◕‿◕｡)",
        ]
        
        # 负面颜文字
        negative_kaomojis = [
            "(╯‵□′)╯︵┻━┻", "(┳Д┳)", "(╥﹏╥)", "(；′⌒`)",
            "(┳◇┳)", "(╯°□°）╯︵ ┻━┻", "(；一_一)", "(¬_¬)",
            "(；￣Д￣）", "(；一_一)", "(¬‿¬)", "(≖_≖ )",
            "(눈_눈)", "(¬▂¬)", "(；一_一)", "(╬ Ò﹏Ó)",
        ]
        
        for km in positive_kaomojis:
            if km in text:
                text = text.replace(km, "[颜文字-正面]")
                kaomoji_sentiment["positive"] += 1
        
        for km in negative_kaomojis:
            if km in text:
                text = text.replace(km, "[颜文字-负面]")
                kaomoji_sentiment["negative"] += 1
        
        # 正则匹配其他颜文字
        for sentiment, pattern in self.kaomoji_patterns.items():
            matches = pattern.findall(text)
            for match in matches:
                text = text.replace(match, f"[颜文字-{sentiment}]")
                kaomoji_sentiment[sentiment] += 1
        
        # 判断整体情感
        if kaomoji_sentiment["negative"] > kaomoji_sentiment["positive"]:
            overall = "negative"
        elif kaomoji_sentiment["positive"] > kaomoji_sentiment["negative"]:
            overall = "positive"
        else:
            overall = "neutral"
        
        total = sum(kaomoji_sentiment.values())
        return text, {"count": total, "sentiment": overall}
    
    def _normalize_slang(self, text: str) -> Tuple[str, List[str]]:
        """游戏黑话标准化"""
        replaced = []
        
        # 按长度降序，避免短词覆盖长词
        sorted_slangs = sorted(self.slang_dict.keys(), key=len, reverse=True)
        
        for slang in sorted_slangs:
            if slang in text:
                standard = self.slang_dict[slang]
                text = text.replace(slang, standard)
                replaced.append(f"{slang}->{standard}")
        
        return text, replaced
    
    def _compress_punctuation(self, text: str) -> str:
        """重复标点压缩"""
        # 多个感叹号压缩
        text = re.sub(r'！{2,}', '[!强调!]', text)
        text = re.sub(r'!{2,}', '[!强调!]', text)
        
        # 多个问号压缩
        text = re.sub(r'？{2,}', '[?疑问?]', text)
        text = re.sub(r'\?{2,}', '[?疑问?]', text)
        
        # 多个省略号压缩
        text = re.sub(r'。{3,}', '[...省略]', text)
        text = re.sub(r'\.{3,}', '[...省略]', text)
        
        # 波浪号处理（表示语气）
        text = re.sub(r'~{2,}', '[~拖音~]', text)
        text = re.sub(r'～{2,}', '[~拖音~]', text)
        
        return text
    
    def _clean_special_chars(self, text: str) -> str:
        """清理特殊字符"""
        # 保留中文、英文、数字、常用标点
        text = re.sub(r'[^\u4e00-\u9fa5a-zA-Z0-9\s\[\]【】！？。，、；：""''（）()]', '', text)
        
        # 压缩多余空格
        text = re.sub(r'\s+', ' ', text)
        
        return text.strip()

import torch
import torch.nn as nn
from torch.utils.data import Dataset, DataLoader
from transformers import (
    BertTokenizer, BertForSequenceClassification,
    TrainingArguments, Trainer,
    EarlyStoppingCallback,
    get_linear_schedule_with_warmup
)
from sklearn.metrics import accuracy_score, precision_recall_fscore_support, classification_report
import numpy as np
import json
from pathlib import Path

class SentimentDataset(Dataset):
    """情感分析数据集"""
    
    def __init__(self, texts, labels, tokenizer, max_length=256):
        self.texts = texts
        self.labels = labels
        self.tokenizer = tokenizer
        self.max_length = max_length
    
    def __len__(self):
        return len(self.texts)
    
    def __getitem__(self, idx):
        text = str(self.texts[idx])
        label = self.labels[idx]
        
        encoding = self.tokenizer(
            text,
            max_length=self.max_length,
            padding='max_length',
            truncation=True,
            return_tensors='pt'
        )
        
        return {
            'input_ids': encoding['input_ids'].flatten(),
            'attention_mask': encoding['attention_mask'].flatten(),
            'labels': torch.tensor(label, dtype=torch.long)
        }


class WeightedTrainer(Trainer):
    """带类别权重的Trainer"""
    
    def __init__(self, class_weights=None, **kwargs):
        super().__init__(**kwargs)
        self.class_weights = class_weights
        if class_weights is not None:
            self.class_weights = torch.tensor(class_weights, dtype=torch.float32)
            if torch.cuda.is_available():
                self.class_weights = self.class_weights.cuda()
    
    def compute_loss(self, model, inputs, return_outputs=False, **kwargs):
        labels = inputs.pop("labels")
        outputs = model(**inputs)
        logits = outputs.logits
        
        if self.class_weights is not None:
            loss_fct = nn.CrossEntropyLoss(weight=self.class_weights)
        else:
            loss_fct = nn.CrossEntropyLoss()
        
        loss = loss_fct(logits, labels)
        
        return (loss, outputs) if return_outputs else loss


def compute_metrics(eval_pred):
    """计算评估指标"""
    predictions, labels = eval_pred
    predictions = np.argmax(predictions, axis=1)
    
    accuracy = accuracy_score(labels, predictions)
    precision, recall, f1, _ = precision_recall_fscore_support(
        labels, predictions, average='weighted', zero_division=0
    )
    
    # 每类详细指标
    class_report = classification_report(
        labels, predictions, 
        target_names=['负面', '中性', '正面'],
        output_dict=True,
        zero_division=0
    )
    
    return {
        'accuracy': accuracy,
        'precision': precision,
        'recall': recall,
        'f1': f1,
        'negative_f1': class_report['负面']['f1-score'],
        'neutral_f1': class_report['中性']['f1-score'],
        'positive_f1': class_report['正面']['f1-score'],
    }


def train_model(train_data_path, val_data_path, config):
    """主训练函数"""
    
    # 加载数据
    with open(train_data_path, 'r', encoding='utf-8') as f:
        train_data = json.load(f)
    with open(val_data_path, 'r', encoding='utf-8') as f:
        val_data = json.load(f)
    
    train_texts = [item['text'] for item in train_data]
    train_labels = [item['label'] for item in train_data]
    val_texts = [item['text'] for item in val_data]
    val_labels = [item['label'] for item in val_data]
    
    # 标签映射
    label_map = {'negative': 0, 'neutral': 1, 'positive': 2}
    train_labels = [label_map[l] for l in train_labels]
    val_labels = [label_map[l] for l in val_labels]
    
    # 加载tokenizer和模型
    tokenizer = BertTokenizer.from_pretrained(config.model_name)
    model = BertForSequenceClassification.from_pretrained(
        config.model_name,
        num_labels=config.num_labels
    )
    
    # 添加自定义词表（游戏术语）
    new_tokens = [
        '原神', '星铁', '绝区零', '崩坏',

        '歪了', '双黄', '氪金', '月卡',
    ]
    tokenizer.add_tokens(new_tokens)
    model.resize_token_embeddings(len(tokenizer))
    
    # 创建数据集
    train_dataset = SentimentDataset(train_texts, train_labels, tokenizer, config.max_length)
    val_dataset = SentimentDataset(val_texts, val_labels, tokenizer, config.max_length)
    
    # 训练参数
    training_args = TrainingArguments(
        output_dir=config.output_dir,
        num_train_epochs=config.num_epochs,
        per_device_train_batch_size=config.batch_size,
        per_device_eval_batch_size=config.batch_size * 2,
        gradient_accumulation_steps=config.gradient_accumulation_steps,
        learning_rate=config.learning_rate,
        weight_decay=config.weight_decay,
        warmup_ratio=config.warmup_ratio,
        
        # 显存优化
        fp16=config.fp16,
        gradient_checkpointing=config.gradient_checkpointing,
        
        # 日志和保存
        logging_dir=f"{config.output_dir}/logs",
        logging_steps=config.logging_steps,
        eval_steps=config.eval_steps,
        save_steps=config.save_steps,
        evaluation_strategy="steps",
        save_strategy="steps",
        load_best_model_at_end=True,
        metric_for_best_model="f1",
        greater_is_better=True,
        
        # 其他
        dataloader_num_workers=4,
        remove_unused_columns=False,
        report_to=["tensorboard"],
    )
    
    # 创建Trainer

    tokenizer.save_pretrained(f"{config.output_dir}/best_model")
    
    # 最终评估
    print("\n最终评估结果:")
    eval_results = trainer.evaluate()
    for key, value in eval_results.items():
        print(f"{key}: {value:.4f}")
    
    return model, tokenizer

# inference/sentiment_analyzer.py
import torch
import torch.nn.functional as F
from transformers import BertTokenizer, BertForSequenceClassification
from typing import Dict, List, Tuple

class GameSentimentAnalyzer:
    """游戏评论情感分析器"""

        return {

                "好", "不错", "满意", "喜欢", "推荐", "值得", "给力",
                "yyds", "永远的神", "绝绝子", "太棒了", "真好"
            ],
            "negative_contexts": [
                "歪", "保底", "非酋", "退钱", "垃圾", "烂", "差", "坑",
                "骗", "骗氪", "逼氪", "退坑", "删游", "卸载", "不玩了",
                "恶心", "失望", "后悔", "亏", "血亏", "坐牢", "痛苦"
            ],
            "sarcasm_patterns": [
                r"可太.*了", r"真是.*啊", r"确实.*呢", r"太.*了呢",
                r".*可真好", r".*真不错", r".*真棒", r".*真厉害",
            ],
            "sentiment_threshold": 0.7
        }
    
    def _load_category_keywords(self) -> Dict[str, List[str]]:
        """加载游戏模块关键词"""
        return {
            "抽卡系统": ["抽卡", "保底", "歪", "小保底", "大保底", "双黄", "十连", "单抽", "up", "池子"],
            "角色强度": ["强度", "t0", "t1", "退环境", "数值膨胀", "超模", "寄了", "仓管", "吃灰"],
            "战斗系统": ["深渊", "12层", "满星", "dps", "伤害", "输出", "辅助", "奶", "t", "eq", "aeq"],
            "养成系统": ["圣遗物", "坐牢", "狗粮", "胚子", "毕业", "小毕业", "大毕业", "强化", "升级"],
            "剧情叙事": ["剧情", "刀", "发刀", "糖", "发糖", "编剧", "文案", "设定", "世界观"],
            "活动设计": ["活动", "版本", "福利", "奖励", "肝", "耗时间", "无聊", "有趣"],
            "运营策略": ["策划", "ch", "运营", "米哈游", "mhy", "补偿", "公告", "调整"],
            "付费体验": ["氪金", "648", "328", "月卡", "白嫖", "骗氪", "逼氪", "性价比"],
        }
    
    def analyze(self, text: str, preprocessed_text: str = None) -> Dict:
        """
        分析单条评论的情感
        
        Args:
            text: 原始文本
            preprocessed_text: 预处理后的文本（可选）
        
        Returns:
            包含情感分析结果的字典
        """
        # 使用预处理文本或原始文本
        input_text = preprocessed_text or text
        
        # 1. 模型推理
        sentiment, confidence, probs = self._model_inference(input_text)
        
        # 2. 反讽检测
        is_sarcasm, sarcasm_reason = self._detect_sarcasm(text, sentiment, confidence)
        
        # 如果检测到反讽且模型判断为正面，则修正为负面
        if is_sarcasm and sentiment == 'positive':
            sentiment = 'negative'
            confidence = min(confidence * 0.8, 0.9)  # 降低置信度
        
        # 3. 情绪强度分级
        intensity = self._calculate_intensity(text, sentiment, confidence)
        
        # 4. 归因分类
        category, keywords = self._categorize(text)
        
        # 5. 生成建议动作
        suggested_action = self._suggest_action(sentiment, intensity, category, is_sarcasm)
        
        return {
            "sentiment": sentiment,
            "sentiment_zh": self.label2zh[sentiment],
            "confidence": round(confidence, 4),
            "intensity": intensity,  # 1-5
            "category": category,
            "keywords": keywords,
            "is_sarcasm": is_sarcasm,
            "sarcasm_reason": sarcasm_reason if is_sarcasm else None,
            "probabilities": {
                self.id2label[i]: round(prob, 4) 
                for i, prob in enumerate(probs)
            },
            "suggested_action": suggested_action,
        }
    
    def _model_inference(self, text: str) -> Tuple[str, float, List[float]]:
        """模型推理"""
        inputs = self.tokenizer(
            text,
            max_length=256,
            padding='max_length',
            truncation=True,
            return_tensors='pt'
        )
        
        inputs = {k: v.to(self.device) for k, v in inputs.items()}
        
        with torch.no_grad():
            outputs = self.model(**inputs)
            probs = F.softmax(outputs.logits, dim=1)[0]
            predicted_id = torch.argmax(probs).item()
            confidence = probs[predicted_id].item()
        
        sentiment = self.id2label[predicted_id]
        return sentiment, confidence, probs.cpu().tolist()
    
    def _detect_sarcasm(self, text: str, sentiment: str, confidence: float) -> Tuple[bool, str]:
        """检测反讽"""
        config = self.sarcasm_config
        
        # 规则1: 正面词+负面语境冲突
        has_positive = any(word in text for word in config["positive_words"])
        has_negative = any(word in text for word in config["negative_contexts"])
        
        if has_positive and has_negative and sentiment == 'positive' and confidence > config["sentiment_threshold"]:
            return True, "正负面词汇冲突，疑似反讽"
        
        # 规则2: 反讽句式
        for pattern in config["sarcasm_patterns"]:
            if re.search(pattern, text) and sentiment == 'positive':
                return True, f"触发反讽句式: {pattern}"
        
        # 规则3: Emoji暗示
        sarcasm_emojis = ['😅', '🙄', '😒', '😐', '🤔']
        if any(emoji in text for emoji in sarcasm_emojis) and sentiment == 'positive':
            return True, "包含反讽暗示Emoji"
        
        return False, ""
    
    def _calculate_intensity(self, text: str, sentiment: str, confidence: float) -> int:
        """计算情绪强度 (1-5)"""
        intensity = 1
        
        # 基于置信度
        if confidence > 0.9:
            intensity += 1
        if confidence > 0.95:
            intensity += 1
        
        # 基于情绪词强度
        strong_negative = ["退坑", "删游", "卸载", "chfm", "骗氪", "逼氪", "恶心", "垃圾"]
        strong_positive = ["yyds", "永远的神", "绝绝子", "完美", "神作"]
        
        if sentiment == 'negative':
            if any(word in text for word in strong_negative):
                intensity = min(5, intensity + 2)
            # 重复标点表示强烈情绪
            if re.search(r'[!！]{2,}', text) or re.search(r'[?？]{2,}', text):
                intensity = min(5, intensity + 1)
        
        elif sentiment == 'positive':
            if any(word in text for word in strong_positive):
                intensity = min(5, intensity + 2)
        
        return intensity
    
    def _categorize(self, text: str) -> Tuple[str, List[str]]:
        """归因分类"""
        category_scores = {}
        matched_keywords = []
        
        for category, keywords in self.category_keywords.items():
            score = 0
            for kw in keywords:
                if kw in text:
                    score += 1
                    matched_keywords.append(kw)
            
            if score > 0:
                category_scores[category] = score
        
        if category_scores:
            best_category = max(category_scores, key=category_scores.get)
        else:
            best_category = "其他"
        
        return best_category, list(set(matched_keywords))
    
    def _suggest_action(self, sentiment: str, intensity: int, category: str, is_sarcasm: bool) -> str:
        """生成建议动作"""
        if sentiment == 'negative':
            if intensity >= 4:
                if is_sarcasm:
                    return "高危：疑似核心玩家反讽，建议人工复核"
                return "高危：强烈负面情绪，建议标记客服跟进"
            elif intensity >= 3:
                return "中危：明显负面情绪，建议关注趋势"
            else:
                return "低危：轻微负面情绪，常规监控"
        
        elif sentiment == 'positive':
            if intensity >= 4:
                return "优质：高度正面反馈，可作为UGC素材"
            return "正面：正常正面反馈"
        
        else:  # neutral
            if category in ["战斗系统", "养成系统"]:
                return "中性：攻略/求助类内容，无需干预"
            return "中性：无明确情绪倾向"
    
    def batch_analyze(self, texts: List[str], batch_size: int = 32) -> List[Dict]:
        """批量分析"""
        results = []
        for i in range(0, len(texts), batch_size):
            batch = texts[i:i+batch_size]
            for text in batch:
                results.append(self.analyze(text))
        return results

"""
性能优化指南

1. 模型推理优化:

3. 批处理优化:
   - 动态batch size根据显存调整
   - 使用DataLoader的多进程加载

4. 并发优化:
   - FastAPI的async/await充分利用异步
   - 使用gunicorn+uvicorn workers

# ONNX导出示例
from transformers import BertForSequenceClassification, BertTokenizer
import torch

def export_to_onnx(model_path, output_path):
    """导出模型为ONNX格式"""
    model = BertForSequenceClassification.from_pretrained(model_path)
    tokenizer = BertTokenizer.from_pretrained(model_path)
    
    # 创建示例输入
    dummy_input = tokenizer(
        "测试文本",
        return_tensors="pt",
        max_length=256,
        padding='max_length',
        truncation=True
    )
    
    # 导出
    torch.onnx.export(
        model,
        (dummy_input['input_ids'], dummy_input['attention_mask']),
        output_path,
        input_names=['input_ids', 'attention_mask'],
        output_names=['output'],
        dynamic_axes={
            'input_ids': {0: 'batch_size', 1: 'sequence'},
            'attention_mask': {0: 'batch_size', 1: 'sequence'},
            'output': {0: 'batch_size'}
        },
        opset_version=14
    )
    print(f"ONNX模型已导出至: {output_path}")


# Redis缓存示例
import redis
import json
import hashlib

class SentimentCache:
    def __init__(self, host='localhost', port=6379, db=0, ttl=3600):
        self.client = redis.Redis(host=host, port=port, db=db)
        self.ttl = ttl
    
    def _get_key(self, text: str) -> str:
        """生成缓存key"""
        return f"sentiment:{hashlib.md5(text.encode()).hexdigest()}"
    
    def get(self, text: str) -> dict:
        """获取缓存结果"""
        key = self._get_key(text)
        cached = self.client.get(key)
        if cached:
            return json.loads(cached)
        return None
    
    def set(self, text: str, result: dict):
        """设置缓存"""
        key = self._get_key(text)
        self.client.setex(key, self.ttl, json.dumps(result))

# monitoring/logger.py
import logging
import json
from datetime import datetime
from pathlib import Path

# 配置日志
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('logs/app.log'),
        logging.StreamHandler()
    ]
)

logger = logging.getLogger(__name__)

class AnalysisLogger:
    """分析日志记录器"""
    
    def __init__(self, log_dir: str = "logs/analysis"):
        self.log_dir = Path(log_dir)
        self.log_dir.mkdir(parents=True, exist_ok=True)
    
    def log_analysis(self, text: str, result: dict, source: str = "api"):
        """记录分析日志"""
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "source": source,
            "text_length": len(text),
            "sentiment": result["sentiment"],
            "confidence": result["confidence"],
            "intensity": result["intensity"],
            "category": result["category"],
            "is_sarcasm": result["is_sarcasm"],
            "processing_time_ms": result.get("processing_time_ms", 0),
        }
        
        # 按日期分文件
        date_str = datetime.now().strftime("%Y-%m-%d")
        log_file = self.log_dir / f"{date_str}.jsonl"
        
        with open(log_file, 'a', encoding='utf-8') as f:
            f.write(json.dumps(log_entry, ensure_ascii=False) + '\n')
    
    def get_daily_stats(self, date: str = None) -> dict:
        """获取每日统计"""
        date = date or datetime.now().strftime("%Y-%m-%d")
        log_file = self.log_dir / f"{date}.jsonl"
        
        if not log_file.exists():
            return {}
        
        stats = {
            "total": 0,
            "sentiment_distribution": {"negative": 0, "neutral": 0, "positive": 0},
            "avg_confidence": 0,
            "avg_processing_time": 0,
            "sarcasm_count": 0,
        }
        
        confidences = []
        processing_times = []
        
        with open(log_file, 'r', encoding='utf-8') as f:
            for line in f:
                entry = json.loads(line)
                stats["total"] += 1
                stats["sentiment_distribution"][entry["sentiment"]] += 1
                confidences.append(entry["confidence"])
                processing_times.append(entry["processing_time_ms"])
                if entry["is_sarcasm"]:
                    stats["sarcasm_count"] += 1
        
        if stats["total"] > 0:
            stats["avg_confidence"] = sum(confidences) / len(confidences)
            stats["avg_processing_time"] = sum(processing_times) / len(processing_times)
        
        return stats
```
